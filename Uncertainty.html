Geographic, Taxonomic, Temporal and Combined Uncertainty

Charlotte R. Patterson

2023-06-27

This code assesses the effect of data uncertainty on potential modelling with the SCAR Antarctic Terrestrial Biodiversity Database. It does so by removing all records that fail an uncertainty criteria and calculating the percentage of records lost during this process. It also assesses the spatial autocorrelation of this uncertainty by calculating Morans I.

Load data

Read in the cleaned biodiversity database (cleaned in file Terrestrial_Antarctic_Biodiversity_Database_Cleaning.Rmd), the cleaned ice-free area vector (cleaned in file ADD_ice_free_area_cleaning.Rmd) and the coastline (Antarctic Digital Database coastline v. 7.5).

Geographic uncertainty

We remove all records without a metric of geographic uncertainty.

# Set areas *without* a metric of geographic uncertainty to 0
Ant_biodf <- Ant_biodf %>% 
  mutate(Geog_Uncertainty = if_else(is.na(coordinateUncertaintyInMetres), 0, 1))

Taxonomic uncertainty

We load the file which identifies the records in the database that have uncertain terms (e.g., sp., c.f.) which was created using the package bdc in the file Terrestrial_Antarctic_Biodiversity_Database_Cleaning.Rmd).

# Load the names of the cleaned data

# Set records *with* taxonomic uncertainty present to 0 (come up as FALSE from bdc_clean function)
Ant_biodf <- Ant_biodf %>% mutate(Taxonomic_Uncertainty = if_else (clean_doc[,4] == FALSE, 0, 1))

Temporal uncertainty

We identify and remove records with no year or date information and remove records from 28/08/2008 because there are > 3000 from the Antarctic Terrestrial Database that have been set as this single date.

Ant_biodf <- Ant_biodf %>% mutate(Temporal_Uncertainty = if_else(is.na(year),0, #if has no year = 0
                                                                 if_else(is.na(Date), 0, #if has no date = 0
                                                                         if_else(Date == "28/08/2008",0, 1))))

Combined uncertainty

Finally, we remove records that violate at least one uncertainty filter.

Ant_biodf <- Ant_biodf %>% mutate(Combined_Uncertainty = if_else(Geog_Uncertainty == 0 | Taxonomic_Uncertainty == 0 | Temporal_Uncertainty == 0, 0, 1))

Geographic uncertainty loss

# Creates a dataframe with lyr.1 = CELL ID, Geographic Uncertainty (0 or 1), then N of records
countgeog <- count(Ant_bio_grid_locs, lyr.1, Geog_Uncertainty)

# Calculates the total number of records for a cell, Geog Uncertainty and Not (0 and 1)
countgeog <- countgeog %>% 
  group_by(lyr.1) %>% 
  mutate(Total_rec = sum(n)) %>% 
  ungroup()

# Set up final dataframe
finalgeog <- data.frame(lyr.1 = numeric(), Perc_Unc = numeric() )

# Now extract the total for each cell ID just once
for(i in unique(countgeog$lyr.1)){ 
  
 filt <- countgeog %>% filter(lyr.1 == i)
  
 filt <- filt %>% mutate(Perc = if_else(nrow(filt) == 1 & Geog_Uncertainty == 1, # If there is 1 row and Geog Uncer is 1 (not present)
                        0, #set to 0%
                        NA))
   
  filt <- filt %>% mutate(Perc = if_else(Geog_Uncertainty == 0, # Otherwise calculate Geog Uncertainty %
                                        n/Total_rec*100,
                                        Perc)) %>% na.omit(.) # Then remove the rows where Geog_Uncertainty = 1 
    
 finalgeog <- finalgeog %>% add_row(lyr.1 = i, Perc_Unc = filt$Perc)

  }


# Full join retains all values, join the grid dataframe and the count values by the cell unique ID

Spatial autocorrelation of uncertainty

Using Morans I from the package spdep, calculate the spatial autocorrelation of the grid cell uncertainty. First we set up the neighbourhood distance to define cells that are considered adjacent. We chose neighbourhood distances for our 100km resolution to be 150km. Neighbours are therefore adjacent by the Queens adjacency rule.

# List the uncertainty rasters
unc_rast_list <- list(geog_unc_rast, taxo_unc_rast, temp_unc_rast)

# Create an index
i <- list(1,2,3)

# Set up the dataframe for saving
MoransI <- data.frame(Uncertainty = numeric(), MoransI = numeric())


MoransI_func <- function(unc, i) {
  
  # Calculating Moran's I  -------------------------------------------------
  
  # Calculate the neighbourhood distance as 1.5 x resolution
  neighbour_dist <- 150000
  
  update_rast_xy <- unc %>% 
    as.data.frame(xy=TRUE) %>% 
    .[, -3] #remove n
  
  print("update_rast_xy created")
  
  # dnearneigh is in metres, finds to index of all values within the specified distance
  nb1 <- dnearneigh(as.matrix(update_rast_xy), 0, neighbour_dist)
  
  # For viewing which neighbours are selected
  # plot(nb1, coords = as.matrix(update_rast_xy))
  
  print("...Nearest neighbours found...")
  
  # Print list of distances
  # dists <- nbdists(nb1, update_rast_xy)
  
  # Generate a weights object
  weights <- nb2listw(nb1, 
                      glist = NULL,
                      style = "B",
                      zero.policy = TRUE) 
  
  print("...Weights calculated...")
  
  # Then extract just the values per cell
  var <- unc %>% 
    as.data.frame() %>% 
    .[,1] 
  
  # Normalize the data so it is not affected by high values
  normalize <- function(x, na.rm = TRUE) {
    return((x- min(x)) /(max(x)-min(x)))
  }
  
  norm_var <- normalize(var)
  
  # Moran's I value for normalized data
  I <- moran(norm_var, 
             weights, 
             zero.policy = T, 
             length(nb1), 
             Szero(weights)) %>% 
    .[[1]]
  
  print("...Moran's I calculated...")
  
  # Add output to dataframe
  MoransI <<- MoransI %>% 
    add_row(Uncertainty = i, MoransI = I)
  
  # Hypothesis test  ----------------------------------------------------
  
  # Run a hypothesis test via permutation
  set.seed(1)
  moran_bootstrap <- moran.mc(var,
                              weights,
                              nsim = 10000,
                              zero.policy = TRUE)
  
  print("...Bootstrapping finished...")
  
  # Print summary outputs  ----------------------------------------------
  
  # Print the output of the hypothesis test
  print(moran_bootstrap)
  
  # Plot the density of expected (under randomisation) vs. observed values of Moran's I 
  plot(moran_bootstrap, main="", las=1)
  
  # Plot the relationship between a cell's value (% records lost from uncertainty) and its nearest neighbour values
  moran.plot(var, weights, zero.policy = T)
  
  # Calculate the number of bins for a correlogram according to Sturge's rule (refer to Borcard et al. 2018)
  no_recs <- unc %>% 
    as.data.frame() %>% 
    nrow()
  
  no_dists <- round(1 + 3.322*log10(no_recs))
  
  # Generate a correlogram to examine the spatial scale of correlation
  subs.correlog <-
    sp.correlogram(nb1, 
                   norm_var, 
                   order = no_dists, 
                   method = "I", 
                   zero.policy = TRUE)
  
  # Print the results with a "Holm" correction for multiple testing
  print(subs.correlog, p.adj.method = "bonferroni")
  
  plot(subs.correlog)
  
}


# Map the correlation function -------------------------------------------------------

map2(unc_rast_list, i, MoransI_func)
